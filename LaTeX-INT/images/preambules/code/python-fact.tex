import timeit

p = 20
def factorial(n):
    result = 1
    for i in range(1, n+1):
        result *= i

    return result


f = factorial(p)
elapsed_time = timeit.timeit(str((factorial(p))), number=10000)

print("Factorial of", p, "=", factorial(p))
print("Time measured :", elapsed_time, "seconds.")
